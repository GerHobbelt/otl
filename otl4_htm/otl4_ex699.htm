OTL 4.0, Example 699 (Simple Insert/Select)

This example demonstrates a simple INSERT with /*+APPEND_VALUES*/ optimizer hint that works the same way as /*+APPEND*/ in INSERT as SELECT statements in the releases of Oracle prior to Oracle 11.2

Source Code

#include <iostream>
using namespace std;

#include <stdio.h>
#define OTL_ORA11G_R2 // Compile OTL 4.0/OCI11.2
#include <otlv4.h> // include the OTL 4.0 header file

otl_connect db; // connect object

void insert()
// insert rows into table
{ 
 otl_stream o(1000, // buffer size
              "insert /*+APPEND_VALUES*/ into test_tab values(:f1<int>,:f2<char[31]>)", 
                 // SQL statement
              db // connect object
             );

// /*+APPEND_VALUES*/ enables OCI direct path data loading feature
// in OCI11.2 and higher

 char tmp[32];

 for(int i=1;i<=100;++i){
  sprintf(tmp,"Name%d",i);
  o<<i<<tmp;
 }
}

int main()
{
 otl_connect::otl_initialize(); // initialize OCI environment
 try{

  db.rlogon("scott/tiger"); // connect to Oracle

  otl_cursor::direct_exec
   (
    db,
    "drop table test_tab",
    otl_exception::disabled // disable OTL exceptions
   ); // drop table

  otl_cursor::direct_exec
   (
    db,
    "create table test_tab(f1 number, f2 varchar2(30))"
    );  // create table

  insert(); // insert records into table

 }

 catch(otl_exception& p){ // intercept OTL exceptions
  cerr<<p.msg<<endl; // print out error message
  cerr<<p.stm_text<<endl; // print out SQL that caused the error
  cerr<<p.var_info<<endl; // print out the variable that caused the error
 }

 db.logoff(); // disconnect from Oracle

 return 0;

}
